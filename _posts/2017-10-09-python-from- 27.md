## Exercise 27: Memorize Logic

```python
# "ド・モルガンの法則"

def not_p_and_q(p, q):
    print("     NOT (P AND Q) = ", not(p and q))

def not_p_or_not_q(p, q):
    print("(NOT P) OR (NOT Q) = ", (not p) or (not q))

print("P = True, Q = True")
not_p_and_q(True, True)
not_p_or_not_q(True, True)

print("P = True, Q = False")
not_p_and_q(True, False)
not_p_or_not_q(True, False)

print("P = False, Q = True")
not_p_and_q(False, True)
not_p_or_not_q(False, True)

print("P = False, Q = False")
not_p_and_q(False, False)
not_p_or_not_q(False, False)
```

## Exercise 29: What If

```python
people = 20
cats = 30
dogs = 15

# Too many cats! The world is doomed!
# The world is dry!
if people < cats:
    print("Too many cats! The world is doomed!")

if people > cats:
    print("Not many cats! The world is saved!")

if people < dogs:
    print("The world is drooled on!")

if people > dogs:
    print("The world is dry!")


dogs += 5
# * dogs = dogs + 5 = 15 + 5 = 20
# →　People == dogs

# People are greater than or equal to dogs.
# People are less than or equal to dogs.
# People are dogs.
if people >= dogs:
    print("People are greater than or equal to dogs.")

if people <= dogs:
    print("People are less than or equal to dogs.")

if people == dogs:
    print("People are dogs.")
``` 

## Exercise 30: Else and If

```python
people = 30
cars = 40
trucks = 15

# We should take the cars.
# * 1番目のif文がTrue
if cars > people:
    print("We should take the cars.")
elif cars < people:
    print("We should not take the cars.")
else:
    print("We can't decide.")
# * elif = ifがFalseの場合に実行される
# * else = 上に記述されている条件式が全てFalseの場合に実行される
# * いずれかの条件式がTrueとなった時にはそれ以降の"elif"文や"else"文は実行されない

# Maybe we could take trucks.
# * 2番目のif文がTrue
if trucks > cars:
    print("That's too many trucks.")
elif trucks < cars:
    print("Maybe we could take trucks.")
else:
    print("We still can't decide.")

# Alright, let's just take the trucks.
# * 1番目のif文がTrue
if people > trucks:
    print("Alright, let's just take the trucks.")
else:
    print("Fine, let's stay home then.")

# * elifで始めることはできない
# * ifとelifは必ずセット！
#
# elif trucks < cars:
#     print("Maybe we could take trucks.")
# else:
#     print("We still can't decide.")
# →
#  elif trucks < cars:
#       ^
# SyntaxError: invalid syntax

# We can't decide.
if cars < people:
    print("We should not take the cars.")
elif cars == people:
    print("We should not take the cars.")
elif cars <= people:
    print("We should not take the cars.")
else:
    print("We can't decide.")
# * elifは何回でも記述できる！
```
